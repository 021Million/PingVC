app.post('/api/request-vc', (req, res) => {
  const { vcId } = req.body;
  const vc = vcs.find(v => v.id === vcId);
  if (!vc) return res.status(404).json({ message: 'VC not found' });

  vc.requestCount = (vc.requestCount || 0) + 1;
  res.json({ success: true, newCount: vc.requestCount });
});
